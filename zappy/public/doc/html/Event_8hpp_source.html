453 bool bindWindowTrigger(const EGE::Event::WindowTrigger &trigger, F &&callback)
454 {
455 switch (trigger)
456 {
457 case EGE::Event::WindowTrigger::WindowClose:
458 GlfwWindowCloseCallbackFunction::getInstance(callback, true);
459 glfwSetWindowCloseCallback(this->_window, [](GLFWwindow *window) {
460 glfwSetWindowShouldClose(window, GL_FALSE);
461 GlfwWindowCloseCallbackFunction::getInstance()->call(window);
462 });
463 return true;
464 case EGE::Event::WindowTrigger::WindowFocus:
465 GlfwWindowFocusCallbackFunction::getInstance(callback, true);
466 glfwSetWindowFocusCallback(this->_window, [](GLFWwindow *window, int focused) {
467 if (focused)
468 GlfwWindowFocusCallbackFunction::getInstance()->call(window, focused);
469 if (!focused)
470 GlfwWindowLostFocusCallbackFunction::getInstance()->call(window, focused);
471 });
472 return true;
473 case EGE::Event::WindowTrigger::WindowLostFocus:
474 GlfwWindowLostFocusCallbackFunction::getInstance(callback, true);
475 glfwSetWindowFocusCallback(this->_window, [](GLFWwindow *window, int focused) {
476 if (focused)
477 GlfwWindowFocusCallbackFunction::getInstance()->call(window, focused);
478 if (!focused)
479 GlfwWindowLostFocusCallbackFunction::getInstance()->call(window, focused);
480 });
481 return true;
482 case EGE::Event::WindowTrigger::WindowFileDrop:
483 GlfwWindowFileDropCallbackFunction::getInstance(callback, true);
484 glfwSetDropCallback(this->_window, [](GLFWwindow *window, int nb, const char* paths[]) {
485 GlfwWindowFileDropCallbackFunction::getInstance()->call(window, nb, paths);
486 });
487 return true;
488 case EGE::Event::WindowTrigger::WindowScroll:
489 GlfwWindowScrollCallbackFunction::getInstance(callback, true);
490 glfwSetScrollCallback(this->_window, [](GLFWwindow *window, double xoffset, double yoffset) {
491 GlfwWindowScrollCallbackFunction::getInstance()->call(window, xoffset, yoffset);
492 });
493 return true;
494 case EGE::Event::WindowTrigger::WindowContentScaleChanged:
495 GlfwWindowContentScaleChangedCallbackFunction::getInstance(callback, true);
496 glfwSetWindowContentScaleCallback(this->_window, [](GLFWwindow *window, float x, float y) {
497 GlfwWindowContentScaleChangedCallbackFunction::getInstance()->call(window, x, y);
498 });
499 return true;
500 case EGE::Event::WindowTrigger::WindowFramebufferResized:
501 GlfwWindowFramebufferResizedCallbackFunction::getInstance(callback, true);
502 glfwSetFramebufferSizeCallback(this->_window, [](GLFWwindow *window, int width, int height) {
503 GlfwWindowFramebufferResizedCallbackFunction::getInstance()->call(window, width, height);
504 });
505 return true;
506 case EGE::Event::WindowTrigger::WindowIconified:
507 GlfwWindowIconifiedCallbackFunction::getInstance(callback, true);
508 glfwSetWindowIconifyCallback(this->_window, [](GLFWwindow *window, int iconified) {
509 GlfwWindowIconifiedCallbackFunction::getInstance()->call(window, iconified);
510 });
511 return true;

513 GlfwWindowMaximizedCallbackFunction::getInstance(callback, true);
514 glfwSetWindowMaximizeCallback(this->_window, [](GLFWwindow *window, int maximized) {
